str += ‘0’;
}
return str;
}
str = str.substring(0, str.lastIndexOf(‘.’) + num + 1);
var start = str.substring(str.lastIndexOf(‘.’) + 1).length;
for (var j = start; j < num; j++) {
str += ‘0’;
}
return str;
};

##什么是模块化
模块化是一种将系统分离成独立功能的方法 ，可将系统分割成独立的功能部分，
严格定义模块接口、模块之间具有透明性。

###模块化的优点
1.代码组织起来更加清晰，提升开发效率
2.减少了意外的bug，便于后期维护
3.避免污染全局作用域，减少了命名冲突

###如何实现模块化
1.将变量写在函数里面
2.将变量使用对象进行管理
3.使用闭包

###模块化规范
1.浏览器端规范
amd：(https://github.com/amdjs/amdjs-api/wiki/AMD) 代表：require.js
cmd: (https://github.com/cmdjs/specification/blob/master/draft/module.md) 代表：sea.js
2.服务器端规范
commonjs:(http://wiki.commonjs.org/wiki/CommonJS) 代表：node.js
ECMAScript2015(ES6)模块化规范

##seajs使用

###cmd规范特性
1.一个模块就是一个单独的文件
2.每个模块都是属于define关键字的回调函数被调用，每个模块都是一个单独的作用域
3.预加载、懒执行(预加载就是要先把所有的模块加载完成，懒执行就是需要执行那个模块才会执行那个模块)

###如何加载模块
1.在html文件中引入sea.js库(假设html同级的js文件夹中存放在js文件)
2.在页面script中使用seajs.use加载主模块，起名一般叫main.js

###创建任务
1.html压缩 htmlmin使用
gulp.task(‘html’,function(){ //task方法是创建任务的实现方式
gulp.src(‘./index.html’)
//src方法获取文件，此处获取同文件夹下的index，html文件
.pipe(htmlmin({collapseWhitespace:true})) //去除html中的空白符
.pipe(gulp.dest(‘./dist’))
//dest方法，指定压缩后的文件存放路径，次例：如果没有dist文件夹会自动创建
.pipe(notify(‘hello html’));//任务完成后提醒插件使用
});
2.rename实例 更改html的名称
gulp.task(‘rename’,function(){
gulp.src(‘./index.html’) //获取文件
.pipe(rename({
direname:”./“, //所在目录
basename:”aloha”, //
prefix:”bonjour-“, //
suffix:”-hola”, //
extname:”.html” // 后缀名
})) //效果生成bonjour-aloha-hola.html文件名
.pipe(gulp.dest(‘./‘)); //更改后文件存放在本目录
});
3.css压缩
gulp.task(‘styles’,function(){
gulp.src(‘./css/.css’) //选择所有css目录下的css文件
.pipe(cssnano()) // 压缩css代码
.pipe(gulp.dest(‘./dist/css’)) // 压缩后保存目录
.pipe(browserSync.stream()) // css改变后，浏览器能收到修改信息
.pipe(notify(‘css complete’)) // 任务完成提醒
});
4.js语法检测、代码合并、压缩
gulp.task(‘scripts’, function() {
//合并操作的方法，此处不执行
//gulp.src([‘./demo.js’,’./index.js’])//要合并操作的文件

gulp.src(‘js/.js’)
.pipe(jshint()) // 进行检查
.pipe(jshint.reporter(‘default’)) // 对代码进行报错提示
.pipe(uglify()) //js代码压缩
.pipe(gulp.dest(‘./dist/js’))
.pipe(browserSync.stream()); //js代码更新，浏览器能收到修改信息
});
5.图片:压缩
gulp.task(‘images’, function() {
gulp.src(‘./images/.‘) //选的images文件夹下的所有图片
.pipe(imagemin())
.pipe(gulp.dest(‘dist/images’))
});
6.主任务
// 主任务
gulp.task(‘default’, [‘html’, ‘styles’, ‘scripts’, ‘images’], function() {
//在执行default任务时，先执行[‘html’, ‘styles’, ‘scripts’, ‘images’]
//的任务，然后再执行下面的代码(称之为任务依赖)
browserSync.init({ //初始化浏览器自动更新操作
server: “./dist” //要管理的文件属于哪个文件夹
});
//当指定的文件发送改变时执行指定任务
gulp.watch(‘./index.html’, [‘html’]);
gulp.watch(‘./css/.css’, [‘styles’]);
gulp.watch(‘./js/.js’, [‘scripts’]);
//当任意html文件发送变化刷新浏览器
gulp.watch(“./*.html”).on(‘change’, browserSync.reload);
});

面向对象笔记

##面向对象性能优化

 1.有关for循环：
 当循环取值时，每次都要重新获取数组长度，会降低代码，尤其是操作HTMLCollections，需要实时查询DOM ，而DOM操作一般是比较昂贵的
 ，因此，我们在循环取值时可以缓存数组或集合的长度，这样只需查询一次，可以提高代码运行效率；
 例：for (var i = 0, max = myarray.length; i < max; i++) { 
     // 使用myarray[i]做点什么}

 2.能力检测：
 在框架加载时判断浏览器是否支持某些功能，保存在变量中，以后使用时判断变量即可，减少原型链的搜索，可以提高性能。
 判断浏览器是否支持某些方法可以进行方法能力检测或方法定义检测。
  方法定义检测：浏览器的内置方法都有类似的结构；{ [native 可以用正则表达式: /\{\s*\[native/ 判断
 例：function querySelectorAll() { [native code] }
 /\{\s*\[native/.test(document.querySelectorAll);   true支持/false不支持

 3.检测是否含有该类名：  cName是要匹配的类名
 将元素和匹配的类名的类名前后都加上一个空格，这样元素中的每一个类名前后都是以空格分开的，如果元素含有要匹配的类名，通过indexOf方法可返回
 一个大于-1的索引，如果没有，就返回-1；
 例：var eles = document.getElementsByTagName('*'),i;
 for(i = 0; i < eles.length; i++){
     if((' ' + eles[i].className + ' ').indexOf(' ' + cName + ' ') >-1){
         alert('匹配成功');
     }else{
         alert('匹配失败');
     }
 }

 4.Object.prototype 的成员
 1) constructor
 2) hasOwnProperty 判断该属性是否为自己提供
 3) propertyIsEnumerable 判断属性是否可以枚举
 4) isPrototypeOf    判断是否为原型对象
 5) toString, toLocaleString, valueOf

 5.documentfragment对象表示内存节点,在页面上不占位置
 创建节点通过  var dfg = document.createDocumentFragment();

 6.添加事件监听
 标准API   dom.addEventListener('click',function(e){...});
 IE低版本  dom.attachEvent('onclick',function(e){...});   该事件执行顺序是倒着的，先添加的会后执行

 7.获取原生的所有事件的方法
 var dv = document.createElement('div'),results = [];
 for(var k in dv){if(k.slice(0,2) == 'on'){results.push(k)}}
 所有的事件名称都保存在results中了。
