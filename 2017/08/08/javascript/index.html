###修改Number的toFixed方法,不四舍五入,直接截取
Number.prototype.toFixed = function (num) {
var str = this.toString();
if (!/./.test(str)) {
// 不是小数
str += ‘.’;
for (var i = 0; i < num; i++) {
str += ‘0’;
}
return str;
}
str = str.substring(0, str.lastIndexOf(‘.’) + num + 1);
var start = str.substring(str.lastIndexOf(‘.’) + 1).length;
for (var j = start; j < num; j++) {
str += ‘0’;
}
return str;
};
