gulp

##使用npm安装指定版本的angular
npm install angular@1.4.7 可以安装指定版本为1.4.7的文件，@后跟版本号

##在gulpfile.js文件中配置gulp
所有引用的gulp插件以及gulp都需要使用npm进行局部安装，然后在根目录创建
gulpfile.js，在文件中配置任务，在本文件夹下打开的命令窗口输入gulp [任务名]
例：gulp script 就可以执行名称为script的任务

###1.引入gulp模块
var gulp = require(‘gulp’);

###2.引入其他gulp插件
1.//引入js压缩插件gulp-uglify
var uglify = require(‘gulp-uglify’);
2.//引入css压缩插件gulp-cssnano
var cssnano = require(‘gulp-cssnano’);
3.//引入html压缩插件
var htmlmin = require(‘gulp-htmlmin’);
4.//引入合并插件gulp-concat
var concat = require(‘gulp-concat’);
5.//引入文件重命名文件gulp-rename
var rename = require(‘gulp-rename’);
6.//引入js代码错误提醒插件
var jshint = require(‘gulp-jshint’);
7.//引入文件重命名插件
var imagemin = require(‘gulp-imagemin’);
8.//引入更改提醒插件
var notify = require(‘gulp-notify’);
9.//引入浏览器自动刷新插件
var browserSync = require(‘browser-sync’).create();
var reload = browserSync.reload;
其它插件：用法见https://www.npmjs.com/search?q=gulp
sass的编译（gulp-sass）
less编译 （gulp-less）
自动添加css前缀（gulp-autoprefixer）
压缩图片（gulp-imagemin）
自动刷新页面（gulp-livereload，谷歌浏览器亲测，谷歌浏览器需安装livereload插件）
图片缓存，只有图片替换了才压缩（gulp-cache）

###创建任务
1.html压缩 htmlmin使用
gulp.task(‘html’,function(){ //task方法是创建任务的实现方式
gulp.src(‘./index.html’)
//src方法获取文件，此处获取同文件夹下的index，html文件
.pipe(htmlmin({collapseWhitespace:true})) //去除html中的空白符
.pipe(gulp.dest(‘./dist’))
//dest方法，指定压缩后的文件存放路径，次例：如果没有dist文件夹会自动创建
.pipe(notify(‘hello html’));//任务完成后提醒插件使用
});
2.rename实例 更改html的名称
gulp.task(‘rename’,function(){
gulp.src(‘./index.html’) //获取文件
.pipe(rename({
direname:”./“, //所在目录
basename:”aloha”, //
prefix:”bonjour-“, //
suffix:”-hola”, //
extname:”.html” // 后缀名
})) //效果生成bonjour-aloha-hola.html文件名
.pipe(gulp.dest(‘./‘)); //更改后文件存放在本目录
});
3.css压缩
gulp.task(‘styles’,function(){
gulp.src(‘./css/.css’) //选择所有css目录下的css文件
.pipe(cssnano()) // 压缩css代码
.pipe(gulp.dest(‘./dist/css’)) // 压缩后保存目录
.pipe(browserSync.stream()) // css改变后，浏览器能收到修改信息
.pipe(notify(‘css complete’)) // 任务完成提醒
});
4.js语法检测、代码合并、压缩
gulp.task(‘scripts’, function() {
//合并操作的方法，此处不执行
//gulp.src([‘./demo.js’,’./index.js’])//要合并操作的文件
//.pipe(concat(‘all.js’)) //合并后名称
gulp.src(‘js/.js’)
.pipe(jshint()) // 进行检查
.pipe(jshint.reporter(‘default’)) // 对代码进行报错提示
.pipe(uglify()) //js代码压缩
.pipe(gulp.dest(‘./dist/js’))
.pipe(browserSync.stream()); //js代码更新，浏览器能收到修改信息
});
5.图片:压缩
gulp.task(‘images’, function() {
gulp.src(‘./images/.‘) //选的images文件夹下的所有图片
.pipe(imagemin())
.pipe(gulp.dest(‘dist/images’))
});
6.主任务
// 主任务
gulp.task(‘default’, [‘html’, ‘styles’, ‘scripts’, ‘images’], function() {
//在执行default任务时，先执行[‘html’, ‘styles’, ‘scripts’, ‘images’]
//的任务，然后再执行下面的代码(称之为任务依赖)
browserSync.init({ //初始化浏览器自动更新操作
server: “./dist” //要管理的文件属于哪个文件夹
});
//当指定的文件发送改变时执行指定任务
gulp.watch(‘./index.html’, [‘html’]);
gulp.watch(‘./css/.css’, [‘styles’]);
gulp.watch(‘./js/.js’, [‘scripts’]);
//当任意html文件发送变化刷新浏览器
gulp.watch(“./*.html”).on(‘change’, browserSync.reload);
});
