文件上传时限制格式和大小

###文件上传时限制格式和大小
function readURL(input, file) {
var img = null;
chksize();
function chksize() {
$(input).addClass(‘upLoad’); //是否有图片的标识
if (input.files && input.files[0]) {
if(input.files[0].size > 5242880){
globalFunc.layerAlert(‘请上传小于5M的图片!’);
$(input).removeClass(‘upLoad’);
$(‘.’ + file).attr(‘src’, ‘’).hide().siblings(‘p’).html(‘请上传身份证照片’).css(‘color’, ‘#4A4A4A’)
.siblings(‘.bg’).css(‘background’, ‘#fff’)
.siblings(‘a’).css(‘background’, ‘url(“../../../statics/img/addIdentify.png”) no-repeat 0 0’);
return false;
}
var reader = new FileReader();
reader.onload = function (e) {
if (img) img.removeNode(true);
img = document.createElement(“img”);
img.style.position = “absolute”;
img.style.visibility = “hidden”;
img.width = 0;
img.height = 0;
$(‘body’).append(img);
img.src = e.target.result;
$(img).on(‘error’,function () {
$(‘.’ + file).attr(‘src’, ‘’).hide().siblings(‘p’).html(‘请上传身份证照片’).css(‘color’, ‘#4A4A4A’)
.siblings(‘.bg’).css(‘background’, ‘#fff’)
.siblings(‘a’).css(‘background’, ‘url(“../../../statics/img/addIdentify.png”) no-repeat 0 0’);
$(input).removeClass(‘upLoad’);
globalFunc.layerAlert(‘您选择的文件不是图片格式!’);
});
$(‘.’ + file).attr(‘src’, e.target.result).show()
.siblings(‘p’).html(‘重新上传’).css(‘color’, ‘#fff’)
.siblings(‘.bg’).css(‘background’, ‘rgba(0,0,0,0.1)’)
.siblings(‘a’).css(‘background’, ‘url(“../../../statics/img/cursorIdentify.png”) no-repeat 0 0’);
};
reader.readAsDataURL(input.files[0]);
}
}
return false;
}
